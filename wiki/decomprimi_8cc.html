                        x->left = temp;

                        if (x->left == NULL){
                                x->carattere = caratteri[conta_car++];
                                x=x->parent;
                                while (x->right != NULL)

                                x->right = temp;
                                x = temp;
                                }
                        else{
                                temp->parent = x;
                                x->right = temp;
                                x=temp;
                                }
                }
        x->carattere = caratteri[conta_car];
        return root;
}

Funzione di Decompressione.

Questa funzione si occupa della decompressione del file. Riceve in input il percorso del file da decomprimere e il nome da dare al file decompresso; Restituisce un valore booleano che indica se l'operazione è andata a buon fine. 

                                                     {
        ifstream f4(sorgente);
        if (!f4)
                return false;

        unsigned char BUFFER = 0;
        int l_buffer = 0;
        pnode root = crea_albero(f4,BUFFER,l_buffer);
        ofstream f5 (destinazione);
        if (!f5)
                return false;
        decodifica (f4, f5, BUFFER, l_buffer, root);
        f4.close();
        f5.close();
        return true;
}
